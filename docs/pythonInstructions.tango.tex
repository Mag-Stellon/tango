\defPython[construire]{{{
def construire(n):
    """Nat -> LISTE[Nat]
        Hypothèse: n > 0
        retourne la liste des entiers naturels 1 à n (inclus)
    """
    l = []
    for i in range(1,n+1):
         l.append(i)
    return l
}}}

Par exemple :

\checkPython{{{
>>> construire(5)
[1, 2, 3, 4, 5]
>>> construire(1)
[1]
}}}
